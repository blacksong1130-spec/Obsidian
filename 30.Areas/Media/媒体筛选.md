const container = dv.el("div","",{cls:"gallery-container"});

async function loadData() {
  const loading = dv.el("div","加载中…",{cls:"loading-indicator"});
  container.appendChild(loading);
  const pages = dv.pages('"02-电影阅读" OR "30.Areas/Media"')
    .filter(p=>p.file)
    .map(p=>{
      let cover = p.image||p.cover||p.poster||p.thumbnail||p.coverImage||"";
      let creator = "";
      if(Array.isArray(p.director)) creator=p.director[0]||"";
      else if(p.director) creator=p.director;
      else if(Array.isArray(p.author)) creator=p.author[0]||"";
      else if(p.author) creator=p.author;
      const score=typeof p.score==='number'?p.score:parseFloat(p.score||0);
      const tags=Array.isArray(p.tags)?p.tags:p.tags?[p.tags]:[];
      return {
        title:p.title||p.file.name, creator, score, type:p.type||"",
        status:p.status||false, cover, path:p.file.path, file:p.file,
        date:p.datePublished||p.releaseDate||p.year||"", tags
      };
    });
  container.removeChild(loading);
  return pages;
}

function saveFilters(f){ try{ localStorage.setItem('bsMediaFilters',JSON.stringify(f)); localStorage.setItem('bsMediaTs',Date.now().toString()); }catch(e){} }
function loadFilters(){ try{ const j=localStorage.getItem('bsMediaFilters'),ts=parseInt(localStorage.getItem('bsMediaTs')||'0'); if(j&&Date.now()-ts<30*60*1000) return JSON.parse(j); }catch(e){} return null; }

function getCoverUrl(cover){
  if(!cover) return '';
  if(cover.startsWith('http')||cover.startsWith('data:')) return cover;
  const clean=cover.replace(/[!\[\]]/g,'').split('|')[0].trim();
  try{ return app.vault.adapter.getResourcePath(clean); }catch(e){ return cover; }
}
function defaultCover(type){
  const colors={movie:'#4a3c6b',teleplay:'#3c4c6b',book:'#5c3c3c',game:'#3c5c3c'};
  const icons={movie:'🎬',teleplay:'📺',book:'📚',game:'🎮'};
  const bg=colors[type]||'#4a4a4a'; const ic=icons[type]||'📄';
  return `https://via.placeholder.com/300x450/${bg.replace('#','')}/ffffff?text=${encodeURIComponent(ic+' '+type)}`;
}
function typeIcon(t){ return {movie:'🎬',teleplay:'📺',book:'📚',game:'🎮'}[t]||'📄'; }
function stars(s){
  s=typeof s==='number'?s:parseFloat(s||0);
  if(s>=9) return '⭐⭐⭐⭐⭐';
  if(s>=7) return '⭐⭐⭐⭐';
  if(s>=5) return '⭐⭐⭐';
  if(s>=3) return '⭐⭐';
  if(s>0) return '⭐';
  return '—';
}
function fmtDate(d){
  if(!d) return '';
  if(/^\d{4}$/.test(d)) return d;
  try{ const dt=new Date(d); if(!isNaN(dt)) return dt.getFullYear()+'-'+(dt.getMonth()+1+'').padStart(2,'0')+'-'+(dt.getDate()+'').padStart(2,'0'); }catch(e){}
  return d;
}

async function updateStatus(page,ns){
  try{
    const content=await app.vault.read(app.vault.getAbstractFileByPath(page.path));
    let nc;
    if(/status:\s*(true|false)/i.test(content)){
      nc=content.replace(/status:\s*(true|false)/i,`status: ${ns}`);
    } else if(content.startsWith('---')){
      const end=content.indexOf('---',3);
      nc=end!==-1?content.slice(0,end)+`status: ${ns}\n`+content.slice(end):content;
    } else { nc=`---\nstatus: ${ns}\n---\n\n`+content; }
    await app.vault.modify(app.vault.getAbstractFileByPath(page.path),nc);
    return true;
  }catch(e){ return false; }
}

function createPagination(total,cur,onChange){
  const p=dv.el("div","",{cls:"pagination"});
  if(cur>1){ const b=dv.el("div","← 上一页",{cls:"page-btn"}); b.addEventListener('click',()=>onChange(cur-1)); p.appendChild(b); }
  for(let i=1;i<=total;i++){
    if(i===1||i===total||(i>=cur-2&&i<=cur+2)){
      const b=dv.el("div",i.toString(),{cls:`page-btn${i===cur?' active':''}`}); b.addEventListener('click',()=>onChange(i)); p.appendChild(b);
    } else if((i===cur-3||i===cur+3)&&p.lastChild&&!p.lastChild.classList.contains('page-btn-ellipsis')){
      p.appendChild(dv.el("div","…",{cls:"page-btn-ellipsis"}));
    }
  }
  if(cur<total){ const b=dv.el("div","下一页 →",{cls:"page-btn"}); b.addEventListener('click',()=>onChange(cur+1)); p.appendChild(b); }
  return p;
}

function renderGallery(pages,filters,curPage,onPageChange){
  let fp=[...pages];
  if(filters.search){
    const terms=filters.search.toLowerCase().split(/\s+/).filter(t=>t.length>0);
    fp=fp.filter(p=>{
      const txt=[p.title,p.creator,p.type,...(p.tags||[])].filter(Boolean).join(' ').toLowerCase();
      return terms.every(t=>txt.includes(t));
    });
  }
  if(filters.type&&filters.type!=='all') fp=fp.filter(p=>p.type===filters.type);
  if(filters.rating&&filters.rating!=='all'){
    const r=parseInt(filters.rating);
    fp=fp.filter(p=>{
      const s=typeof p.score==='number'?p.score:parseFloat(p.score||0);
      if(r===5) return s>=9; if(r===4) return s>=7&&s<9; if(r===3) return s>=5&&s<7;
      if(r===2) return s>=3&&s<5; if(r===1) return s>0&&s<3; return false;
    });
  }
  if(filters.status&&filters.status!=='all') fp=fp.filter(p=>String(p.status)===filters.status);
  if(filters.sort){
    fp=fp.sort((a,b)=>{
      let va,vb;
      if(filters.sort.f==='score'){ va=typeof a.score==='number'?a.score:parseFloat(a.score||0); vb=typeof b.score==='number'?b.score:parseFloat(b.score||0); }
      else if(filters.sort.f==='title'){ va=(a.title||'').toLowerCase(); vb=(b.title||'').toLowerCase(); }
      else if(filters.sort.f==='date'){ va=a.date?new Date(a.date):new Date(0); vb=b.date?new Date(b.date):new Date(0); }
      else { va=a[filters.sort.f]||''; vb=b[filters.sort.f]||''; }
      return filters.sort.d==='asc'?(va>vb?1:-1):(va<vb?1:-1);
    });
  }

  const total=Math.max(1,Math.ceil(fp.length/ITEMS_PER_PAGE));
  const vp=Math.min(total,Math.max(1,curPage));
  savePage(vp);
  const slice=fp.slice((vp-1)*ITEMS_PER_PAGE,vp*ITEMS_PER_PAGE);

  // 更新 header 计数
  const hdr=document.querySelector('.gallery-header .gallery-count');
  if(hdr) hdr.textContent=`共 ${fp.length} 项`;

  // 清空网格
  const grid=document.querySelector('.gallery-grid');
  grid.innerHTML='';

  if(fp.length===0){ grid.innerHTML='<div class="no-results">没有找到匹配的结果</div>'; return {total,cur:1}; }

  slice.forEach(page=>{
    const card=dv.el("div","",{cls:"gallery-card"});
    const img=document.createElement("img");
    img.src=page.cover?getCoverUrl(page.cover):defaultCover(page.type);
    img.alt=page.title;
    img.onerror=()=>{ img.src=defaultCover(page.type); };
    card.appendChild(img);
    if(page.type){ const tb=dv.el("div",typeIcon(page.type)+" "+page.type,{cls:"type-badge"}); card.appendChild(tb); }
    if(page.score){ const rb=dv.el("div",stars(page.score),{cls:"rating-badge"}); card.appendChild(rb); }
    const df=fmtDate(page.date);
    if(df){ const db=dv.el("div",df,{cls:"date-badge"}); card.appendChild(db); }
    const sb=dv.el("div",page.status?"✓ 已完成":"○ 未完成",{cls:"status-badge"});
    sb.addEventListener('click',async e=>{
      e.stopPropagation();
      const ns=!page.status;
      if(await updateStatus(page,ns)){ page.status=ns; sb.textContent=ns?"✓ 已完成":"○ 未完成"; }
    });
    card.appendChild(sb);
    card.addEventListener('click',()=>app.workspace.openLinkText(page.path,'',false));
    grid.appendChild(card);
  });

  // 分页
  const oldPag=document.querySelector('.pagination');
  if(oldPag) container.removeChild(oldPag);
  if(total>1){ container.appendChild(createPagination(total,vp,p=>{ renderGallery(pages,filters,p,onPageChange); })); }

  return {total,cur:vp};
}

async function init(){
  const grid=dv.el("div","",{cls:"gallery-grid"});

  const hdr=dv.el("div","",{cls:"gallery-header"});
  hdr.appendChild(dv.el("div","",{cls:"gallery-count"}));
  container.appendChild(hdr);

  const pages=await loadData();
  const saved=loadFilters()||{search:'',type:'all',rating:'all',status:'all',sort:{f:'score',d:'desc'}};
  let curPage=loadPage();

  const bar=createFilterBar(pages,newF=>{
    Object.assign(saved,newF);
    curPage=1; savePage(1);
    renderGallery(pages,saved,curPage,()=>{});
  });
  container.appendChild(bar);
  container.appendChild(grid);
  renderGallery(pages,saved,curPage,()=>{});
}

dv.container.appendChild(container);
